198 private void EventRun()
199 {
200 bool isAllSuccess = true;
201
203 foreach(KeyValuePair<string, IEvent> tuple in m_table)
204 {
205 IEvent eventInterface = tuple.Value;
206 bool isBreakCall = InvokeByEventInterface(eventInterface, ref isAllSuccess);
207
208 if(eventInterface.RemoveThis)
209 {
210 m_removeEventNameList.Add(tuple.Key);
211 }
212
213 if (isBreakCall) break;
214 }
215
217 if(m_haveEventSO && m_eventSO.AllModifierIsTrueWhenExecute)
218 {
219 TryAllInvoke(isAllSuccess);
220 }
221
222 TryRemoveEvent();
223 }

230 private bool InvokeByEventInterface(IEvent eventInterface,
231 ref bool isAllSuccess)
232 {
233 eventInterface.ModifierInvoke(GameObject, m_table);
234
236 if(eventInterface.ReturnValue)
237 {
239 if(m_haveEventSO ? !m_eventSO.AllModifierIsTrueWhenExecute : true)
240 {
241 eventInterface.ContentInvoke(GameObject, m_table);
242 }
243 }
245 else
246 {
247 isAllSuccess = false;
249 if(m_haveEventSO && !m_eventSO.AllModifierIsTrueWhenExecute)
250 {
251 eventInterface.NotModifierContentInvoke(GameObject, m_table);
252 }
253
255 if(m_haveEventSO && m_eventSO.FinishIfFalse)
256 {
257 return true;
258 }
259 }
260
261 return false;
262 }
