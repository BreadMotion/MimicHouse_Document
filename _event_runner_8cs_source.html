179 private void EventRun()
180 {
181 void BranchByModifierTrue(bool value, bool allSuccess)
182 {
183 //全プロセス成功時の時にのみ実行される場合
184 if(value) TryAllInvoke(allSuccess);
185 }
186
187 bool isAllSuccess = true;
188 foreach(var table in m_table)
189 {//登録した全イベント処理を実行
190 var eventInterface = table.Value;
191 bool isBreakCall = InvokeByEventInterface(eventInterface, ref isAllSuccess);
192 if(isBreakCall) break;
193 }
194
195 bool allModifierIsTrueWhenExecute = m_haveEventSO ?
196 m_eventSO.AllModifierIsTrueWhenExecute :
197 m_eventData.GetAllModifierIfTrueWhenExecute();
198
199 BranchByModifierTrue(allModifierIsTrueWhenExecute, isAllSuccess);
200 }

209 private bool InvokeByEventInterface(IEvent eventInterface, ref bool isAllSuccess)
210 {
211 eventInterface.ModifierInvoke(gameObject, m_table);
212
213 if(eventInterface.ReturnValue)
214 {//イベント実行結果がTrueの場合
215 if(!m_eventSO.AllModifierIsTrueWhenExecute)
216 {//各プロセスの条件がTrueじゃなくても動作する場合
217 eventInterface.ContentInvoke(gameObject, m_table);
218 }
219 }
220 else
221 {//イベント実行結果がFalseの場合
222 isAllSuccess = false;
223 if(!m_eventSO.AllModifierIsTrueWhenExecute)
224 {//各プロセスの条件がTrueじゃなくても動作する場合
225 eventInterface.NotModifierContentInvoke(gameObject, m_table);
226 }
227
228 if(m_eventSO.FinishIfFalse)
229 {//プロセスにFalseがあったら強制終了する場合
230 return true;
231 }
232 }
233 return false;
234 }
