197 private void EventRun()
198 {
199 bool isAllSuccess = true;
200 foreach (var tuple in m_table)
201 {//登録した全イベント処理を実行
202 var eventInterface = tuple.Value;
203 bool isBreakCall = InvokeByEventInterface(eventInterface, ref isAllSuccess);
204 if(eventInterface.RemoveThis) m_removeEventNameList.Add(tuple.Key);
205 if (isBreakCall) break;
206 }
207
208 //全てのイベントがTrueだったか
209 bool allModifierIsTrueWhenExecute = m_haveEventSO ?
210 m_eventSO.AllModifierIsTrueWhenExecute :
211 false;
212
213 if(allModifierIsTrueWhenExecute) TryAllInvoke(isAllSuccess);
214 TryRemoveEvent();
215 }

222 private bool InvokeByEventInterface(IEvent eventInterface,
223 ref bool isAllSuccess)
224 {
225 eventInterface.ModifierInvoke(gameObject, m_table);
226
227 if (eventInterface.ReturnValue)
228 {//イベント実行結果がTrueの場合
229 if (m_haveEventSO ? !m_eventSO.AllModifierIsTrueWhenExecute : true)
230 {//各プロセスの条件がTrueじゃなくても動作する場合
231 eventInterface.ContentInvoke(gameObject, m_table);
232 }
233 }
234 else
235 {//イベント実行結果がFalseの場合
236 isAllSuccess = false;
237 if (m_haveEventSO ? !m_eventSO.AllModifierIsTrueWhenExecute : false)
238 {//各プロセスの条件がTrueじゃなくても動作する場合
239 eventInterface.NotModifierContentInvoke(gameObject, m_table);
240 }
241
242 if (m_haveEventSO ? m_eventSO.FinishIfFalse : false)
243 {//プロセスにFalseがあったら強制終了する場合
244 return true;
245 }
246 }
247 return false;
248 }
