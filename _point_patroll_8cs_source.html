195 [ContextMenu("パトロールポイントの接続")]
196 public void ExecBuild()
197 {
199 m_nextPoints.Clear();
200 m_prePoint = null;
201 m_wasBuildComplete = false;
202
203 PointPatroll startPoint = null;
204 if (this.m_pointType == PointType.START) startPoint = this;
205 else startPoint = this.transform.parent.GetComponent<PointPatroll>();
206
208 var patrollPoints = startPoint.transform.GetComponentsInChildren<PointPatroll>();
209 if (patrollPoints.Length <= 0)
210 {
211 Debug.LogWarning("子オブジェクトにポイントオブジェクトがありません");
212 }
213
215 int startPointIdx = -1;
216 int endPointIdx = -1;
217 for (var i = 0; i < patrollPoints.Length; i++)
218 {
219 var point = patrollPoints[i];
220 point.m_nextPoints.Clear();
221 point.m_prePoint = null;
222 point.m_wasBuildComplete = false;
223
224 if (point.m_pointType == PointType.START)
225 {
226 startPointIdx = i;
227 }
228 else if (point.m_pointType == PointType.END)
229 {
230 endPointIdx = i;
231 point.transform.parent = startPoint.transform;
232 point.m_prePoint = startPoint;
233 }
234 else
235 {
236 point.transform.parent = startPoint.transform;
237 point.m_prePoint = startPoint;
238 }
239 }
240
242 if (startPointIdx == -1)
243 {
244 Debug.LogWarning("スタートポイントが設定されていません。");
245 return;
246 }
247 if (endPointIdx == -1)
248 {
249 Debug.LogWarning("エンドポイントが設定されていません。");
250 return;
251 }
252
254 var parentGo = patrollPoints[startPointIdx].transform.parent;
255 for (var i = 0; i < patrollPoints.Length; i++)
256 {
257 var point = patrollPoints[i];
258 point.m_wasBuildComplete = true;
259 if (point.transform == parentGo.parent) continue;
260
261 for (var j = 0; j < patrollPoints.Length; j++)
262 {
263 var sourcePoint = patrollPoints[j];
264 if (sourcePoint.transform == parentGo) continue;
265 if (j == i) continue;
266
267 point.AddNextPoint(sourcePoint);
268 }
269 }
270
272 m_wasBuildComplete = true;
273 }
