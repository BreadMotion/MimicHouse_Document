201 public void ExecBuild()

205 m_prePoint = null;
206 m_wasBuildComplete = false;
207
208 PointPatroll startPoint = null;
209 if (this.m_pointType == PointType.START) startPoint = this;
210 else startPoint = this.transform.parent.GetComponent<PointPatroll>();
211
213 var patrollPoints = startPoint.transform.GetComponentsInChildren<PointPatroll>();
214 if (patrollPoints.Length <= 0)
215 {
216 Debug.LogWarning("子オブジェクトにポイントオブジェクトがありません");
217 }
218
220 int startPointIdx = -1;
221 int endPointIdx = -1;
222 for (var i = 0; i < patrollPoints.Length; i++)
223 {
224 var point = patrollPoints[i];
225 point.m_nextPoints.Clear();
226 point.m_prePoint = null;
227 point.m_wasBuildComplete = false;
228
229 if (point.m_pointType == PointType.START)
230 {
231 startPointIdx = i;
232 }
233 else if (point.m_pointType == PointType.END)
234 {
235 endPointIdx = i;
236 point.transform.parent = startPoint.transform;
237 point.m_prePoint = startPoint;
238 }
239 else
240 {
241 point.transform.parent = startPoint.transform;
242 point.m_prePoint = startPoint;
243 }
244 }
245
247 if (startPointIdx == -1)
248 {
249 Debug.LogWarning("スタートポイントが設定されていません。");
250 return;
251 }
252 if (endPointIdx == -1)
253 {
254 Debug.LogWarning("エンドポイントが設定されていません。");
255 return;
256 }
257
259 var parentGo = patrollPoints[startPointIdx].transform.parent;
260 for (var i = 0; i < patrollPoints.Length; i++)
261 {
262 var point = patrollPoints[i];
263 point.m_wasBuildComplete = true;
264 if (point.transform == parentGo.parent) continue;
265
266 for (var j = 0; j < patrollPoints.Length; j++)
267 {
268 var sourcePoint = patrollPoints[j];
269 if (sourcePoint.transform == parentGo) continue;
270 if (j == i) continue;
271
272 point.AddNextPoint(sourcePoint);
273 }
274 }
275
277 m_wasBuildComplete = true;
