135 private void LoadItemSO(int layerIndex,
136 string name)
137 {
138 //アイテムスロットをコンテキストから取得する
139 var slotHandle = Addressables.LoadAssetAsync<SO.ItemSlotScriptableObject>(name + "SlotData");
140
141 //ロード完了時の処理を登録
142 slotHandle.Completed +=
143 handle =>
144 {
145 //イメージの初期化
146 var newObject = Instantiate(m_prefabBaseSlot);
147 var sprite = newObject.GetComponent<Image>();
148 sprite.sprite = handle.Result.SlotTexture;
149
150 //トランスフォームの初期化
151 newObject.transform.SetParent(this.transform);
152 newObject.transform.localPosition = new Vector3(FirstDiffX + PartialDiffX * layerIndex, PartialDiffX, 0.0f);
153 newObject.transform.localScale = Vector3.one * SizeMag;
154
155 //アイテムスロットコンポーネントの初期化＆管理対象として保持
156 var newObjectItemSlot = newObject.GetComponent<App.ItemSlot>();
157 newObjectItemSlot.SetMustData((ItemLayer)layerIndex, handle.Result.ItemTexture);
158 m_slot.Add(newObjectItemSlot);
159
160 //テクスチャを管理対象として保持
161 m_itemSlotTex.Add((ItemLayer)layerIndex, handle.Result.ItemTexture);
162
163 //サイズ初期化
164 var rect = newObject.GetComponent<RectTransform>();
165 m_slotWidth = rect.rect.width;
166 w_slotHeight = rect.rect.height;
167 };
168
169 //同期させる
170 slotHandle.WaitForCompletion();
171 }
