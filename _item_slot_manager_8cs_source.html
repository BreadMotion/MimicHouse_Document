144 private void LoadItemSO(int layerIndex,
145 string name)
146 {
147 //アイテムスロットをコンテキストから取得する
148 var slotHandle = Addressables.LoadAssetAsync<ItemSlotScriptableObject>(name + "SlotData");
149
150 //ロード完了時の処理を登録
151 slotHandle.Completed +=
152 handle =>
153 {
154 //イメージの初期化
155 var newObject = Instantiate(m_prefabBaseSlot);
156 var sprite = newObject.GetComponent<Image>();
157 sprite.sprite = handle.Result.SlotTexture;
158
159 //トランスフォームの初期化
160 newObject.transform.SetParent(Transform);
161 newObject.transform.localPosition = new Vector3(FirstDiffX + PartialDiffX * layerIndex, PartialDiffX, 0.0f);
162 newObject.transform.localScale = Vector3.one * SizeMag;
163
164 //アイテムスロットコンポーネントの初期化＆管理対象として保持
165 var newObjectItemSlot = newObject.GetComponent<ItemSlot>();
166 newObjectItemSlot.SetMustData((ItemLayer)layerIndex, handle.Result.ItemTexture);
167 m_slot.Add(newObjectItemSlot);
168
169 //テクスチャを管理対象として保持
170 m_itemSlotTex.Add((ItemLayer)layerIndex, handle.Result.ItemTexture);
171
172 //サイズ初期化
173 var rect = newObject.GetComponent<RectTransform>();
174 m_slotWidth = rect.rect.width;
175 w_slotHeight = rect.rect.height;
176 };
177
178 //同期させる
179 slotHandle.WaitForCompletion();
180 }
