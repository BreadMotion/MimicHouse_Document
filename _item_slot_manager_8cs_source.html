113 private void LoadItemSO(int layerIndex, string name)
114 {
115 //アイテムスロットをコンテキストから取得する
116 var slotHandle = Addressables.LoadAssetAsync<SO.ItemSlotScriptableObject>(name + "SlotData");
117
118 //ロード完了時の処理を登録
119 slotHandle.Completed +=
120 handle =>
121 {
122 //イメージの初期化
123 var newObject = Instantiate(m_prefabBaseSlot);
124 var sprite = newObject.GetComponent<Image>();
125 sprite.sprite = handle.Result.SlotTexture;
126
127 //トランスフォームの初期化
128 newObject.transform.SetParent(this.transform);
129 newObject.transform.localPosition = new Vector3(FirstDiffX + PartialDiffX * layerIndex, PartialDiffX, 0.0f);
130 newObject.transform.localScale = Vector3.one * SizeMag;
131
132 //アイテムスロットコンポーネントの初期化＆管理対象として保持
133 var newObjectItemSlot = newObject.GetComponent<App.ItemSlot>();
134 newObjectItemSlot.SetMustData((ItemLayer)layerIndex, handle.Result.ItemTexture);
135 m_slot.Add(newObjectItemSlot);
136
137 //テクスチャを管理対象として保持
138 m_itemSlotTex.Add((ItemLayer)layerIndex, handle.Result.ItemTexture);
139
140 //サイズ初期化
141 var rect = newObject.GetComponent<RectTransform>();
142 m_slotWidth = rect.rect.width;
143 w_slotHeight = rect.rect.height;
144 };
145
146 //同期させる
147 slotHandle.WaitForCompletion();
148 }
