36 public IItem GenerateItem(ItemLayer itemLayer)
37 {
38 if (m_itemContentList.Count != m_itemList.Count)
39 {//アイテムコンテキスト数とアイテム数が揃ってないとバグるのでログ出力
40 Debug.LogWarning("コンテキストと実アイテムオブジェクトの数があっていません。バグもしくはエラーの発生原因となります。");
41 }
42
43 var itemLayerIndex = (int)itemLayer;
44 var itemScriptable = m_itemContentList[itemLayerIndex];
45 var item = m_itemList[itemLayerIndex];
46
47 //アイテムの総数をUISlotManagerに通達する
48 if (itemScriptable.MaxCount > item.NowCount)
49 {
50 item.AddNowCount();
51 }
52
53 ItemSlotManager.I.SetItemCountText(item.ItemLayer, item.NowCount.ToString());
54 return item;
55 }
